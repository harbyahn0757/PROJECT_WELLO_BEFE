    <script>
        const openModalBtn = document.getElementById('openModalBtn');
        const closeModalBtn = document.getElementById('closeModalBtn');
        const getRecommendationBtn = document.getElementById('getRecommendationBtn');
        const aiModal = document.getElementById('aiModal');
        const modalForm = document.getElementById('modalForm');
        const modalResult = document.getElementById('modalResult');
        const loader = document.getElementById('loader');
        const resultText = document.getElementById('resultText');

        openModalBtn.addEventListener('click', () => {
            aiModal.classList.remove('opacity-0', 'pointer-events-none');
            aiModal.querySelector('.modal-content').classList.remove('scale-95');
            // Reset modal to form view
            modalForm.classList.remove('hidden');
            modalResult.classList.add('hidden');
            resultText.innerHTML = '';
        });

        closeModalBtn.addEventListener('click', () => {
            aiModal.classList.add('opacity-0', 'pointer-events-none');
            aiModal.querySelector('.modal-content').classList.add('scale-95');
        });

        getRecommendationBtn.addEventListener('click', () => {
            const age = document.getElementById('age').value;
            const gender = document.querySelector('input[name="gender"]:checked');
            const concerns = document.getElementById('concerns').value;

            if (!age || !gender) {
                alert('나이와 성별을 입력해주세요.');
                return;
            }

            modalForm.classList.add('hidden');
            modalResult.classList.remove('hidden');
            loader.classList.remove('hidden');
            resultText.classList.add('hidden');

            callGeminiApi(age, gender.value, concerns);
        });
        
        async function callGeminiApi(age, gender, concerns) {
            const apiKey = ""; // API key is handled by the execution environment.
            const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-05-20:generateContent?key=${apiKey}`;

            const systemPrompt = `당신은 '김현우내과의원'의 친절하고 전문적인 AI 헬스 어드바이저입니다. 사용자의 정보를 바탕으로 '국가검진', '국가 암검진', '종합검진' 중에서 가장 적합한 검진 프로그램을 추천하고, 그 이유를 이해하기 쉽게 설명해주세요. 답변은 항상 한국어로, 존댓말로 작성해야 합니다. 추천 내용은 일반적인 건강 정보에 기반하며, 의학적 진단이 아님을 명확히 밝혀주세요.`;
            
            const userQuery = `
                - 나이: ${age}세
                - 성별: ${gender}
                - 평소 건강 고민: ${concerns || "특별한 내용 없음"}
            `;

            const payload = {
                systemInstruction: {
                    parts: [{ text: systemPrompt }]
                },
                contents: [{
                    parts: [{ text: userQuery }]
                }],
            };

            try {
                // Exponential backoff for retries
                let response;
                let retries = 3;
                let delay = 1000;
                for (let i = 0; i < retries; i++) {
                    response = await fetch(apiUrl, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify(payload)
                    });
                    if (response.status === 200) break;
                    if (response.status === 429 || response.status >= 500) {
                        await new Promise(resolve => setTimeout(resolve, delay));
                        delay *= 2;
                    } else {
                        break; // Don't retry for other client-side errors
                    }
                }


                if (!response.ok) {
                    throw new Error(`API 요청 실패: ${response.status}`);
                }
                const result = await response.json();
                const candidate = result.candidates?.[0];

                if (candidate && candidate.content?.parts?.[0]?.text) {
                    const text = candidate.content.parts[0].text;
                    resultText.textContent = text;
                } else {
                    resultText.textContent = '결과를 생성하는 데 실패했습니다. 잠시 후 다시 시도해주세요.';
                }

            } catch (error) {
                console.error('Gemini API 호출 중 오류 발생:', error);
                resultText.textContent = 'AI 추천을 받는 중 오류가 발생했습니다. 잠시 후 다시 시도해주세요.';
            } finally {
                loader.classList.add('hidden');
                resultText.classList.remove('hidden');
            }
        }

    </script>
