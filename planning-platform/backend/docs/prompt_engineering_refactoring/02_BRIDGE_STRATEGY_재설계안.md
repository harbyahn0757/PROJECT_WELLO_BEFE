### 추가 전략: 조건부 활성화

#### 논리 설계
```python
def select_bridge_strategies(patient_context: Dict) -> List[Dict]:
    """환자 상황에 맞는 Bridge Strategy 동적 선택"""
    
    selected = []
    
    # 1. 만성질환 우선
    if patient_context.get('blood_pressure') == 'abnormal':
        selected.append(BRIDGE_STRATEGIES['cardiovascular'])  # 혈관 건강
    
    if patient_context.get('blood_sugar') == 'abnormal':
        selected.append(BRIDGE_STRATEGIES['metabolic'])  # 대사 건강
    
    if patient_context.get('liver') == 'abnormal':
        selected.append(BRIDGE_STRATEGIES['liver'])  # 간 건강
    
    # 2. 가족력 기반
    family_history = patient_context.get('family_history', [])
    if 'diabetes' in family_history or 'hypertension' in family_history:
        selected.append(BRIDGE_STRATEGIES['kidney'])  # 신장 기능
    
    # 3. 암은 조건부만
    if patient_context.get('smoking_years', 0) >= 30:
        selected.append(BRIDGE_STRATEGIES['lung_cancer'])  # 폐암
    
    if 'stomach_cancer' in family_history:
        selected.append(BRIDGE_STRATEGIES['stomach_cancer'])  # 위암
    
    # 4. 데이터 공백 시
    if patient_context.get('checkup_gap_years', 0) >= 3:
        selected.append(BRIDGE_STRATEGIES['baseline_data'])  # 베이스라인
    
    return selected[:6]  # 최대 6개
```
